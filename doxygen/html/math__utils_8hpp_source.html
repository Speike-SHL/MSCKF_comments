24 inline Eigen::Matrix3d skewSymmetric(const Eigen::Vector3d &w)
25 {
26  Eigen::Matrix3d w_hat;
27  w_hat(0, 0) = 0;
28  w_hat(0, 1) = -w(2);
29  w_hat(0, 2) = w(1);
30  w_hat(1, 0) = w(2);
31  w_hat(1, 1) = 0;
32  w_hat(1, 2) = -w(0);
33  w_hat(2, 0) = -w(1);
34  w_hat(2, 1) = w(0);
35  w_hat(2, 2) = 0;
36  return w_hat;
37 }

54 inline Eigen::Vector4d quaternionMultiplication(
55  const Eigen::Vector4d &q1,
56  const Eigen::Vector4d &q2)
57 {
58  Eigen::Matrix4d L;
59  L(0, 0) = q1(3);
60  L(0, 1) = q1(2);
61  L(0, 2) = -q1(1);
62  L(0, 3) = q1(0);
63  L(1, 0) = -q1(2);
64  L(1, 1) = q1(3);
65  L(1, 2) = q1(0);
66  L(1, 3) = q1(1);
67  L(2, 0) = q1(1);
68  L(2, 1) = -q1(0);
69  L(2, 2) = q1(3);
70  L(2, 3) = q1(2);
71  L(3, 0) = -q1(0);
72  L(3, 1) = -q1(1);
73  L(3, 2) = -q1(2);
74  L(3, 3) = q1(3);
75 
76  Eigen::Vector4d q = L * q2;
77  quaternionNormalize(q);
78  return q;

91 inline Eigen::Vector4d smallAngleQuaternion(
92  const Eigen::Vector3d &dtheta)
93 {
94  // δq ~= (1/2δθ, 1)
95  Eigen::Vector3d dq = dtheta / 2.0;
96  Eigen::Vector4d q;
97  double dq_square_norm = dq.squaredNorm();
98 
99  // 这么做就是为了符合四元数的定义
100  // q(3)的平方+q.head<3>()的平方和的和是1
101  if (dq_square_norm <= 1)
102  {
103  q.head<3>() = dq;
104  q(3) = std::sqrt(1 - dq_square_norm);
105  }
106  else
107  {
108  q.head<3>() = dq;
109  q(3) = 1;
110  q = q / std::sqrt(1 + dq_square_norm);
111  }
112 
113  return q;
114 }

126 inline Eigen::Matrix3d quaternionToRotation(
127  const Eigen::Vector4d &q)
128 {
129  const Eigen::Vector3d &q_vec = q.block(0, 0, 3, 1);
130  const double &q4 = q(3);
131  Eigen::Matrix3d R =
132  (2 * q4 * q4 - 1) * Eigen::Matrix3d::Identity() -
133  2 * q4 * skewSymmetric(q_vec) +
134  2 * q_vec * q_vec.transpose();
135  // TODO: Is it necessary to use the approximation equation
136  // (Equation (70)) when the rotation angle is small?
137  return R;
138 }

150 inline Eigen::Vector4d rotationToQuaternion(
151  const Eigen::Matrix3d &R)
152 {
153  Eigen::Vector4d score;
154  score(0) = R(0, 0);
155  score(1) = R(1, 1);
156  score(2) = R(2, 2);
157  score(3) = R.trace();
158 
159  int max_row = 0, max_col = 0;
160  score.maxCoeff(&max_row, &max_col);
161 
162  Eigen::Vector4d q = Eigen::Vector4d::Zero();
163  if (max_row == 0)
164  {
165  q(0) = std::sqrt(1 + 2 * R(0, 0) - R.trace()) / 2.0;
166  q(1) = (R(0, 1) + R(1, 0)) / (4 * q(0));
167  q(2) = (R(0, 2) + R(2, 0)) / (4 * q(0));
168  q(3) = (R(1, 2) - R(2, 1)) / (4 * q(0));
169  }
170  else if (max_row == 1)
171  {
172  q(1) = std::sqrt(1 + 2 * R(1, 1) - R.trace()) / 2.0;
173  q(0) = (R(0, 1) + R(1, 0)) / (4 * q(1));
174  q(2) = (R(1, 2) + R(2, 1)) / (4 * q(1));
175  q(3) = (R(2, 0) - R(0, 2)) / (4 * q(1));
176  }
177  else if (max_row == 2)
178  {
179  q(2) = std::sqrt(1 + 2 * R(2, 2) - R.trace()) / 2.0;
180  q(0) = (R(0, 2) + R(2, 0)) / (4 * q(2));
181  q(1) = (R(1, 2) + R(2, 1)) / (4 * q(2));
182  q(3) = (R(0, 1) - R(1, 0)) / (4 * q(2));
183  }
184  else
185  {
186  q(3) = std::sqrt(1 + R.trace()) / 2.0;
187  q(0) = (R(1, 2) - R(2, 1)) / (4 * q(3));
188  q(1) = (R(2, 0) - R(0, 2)) / (4 * q(3));
189  q(2) = (R(0, 1) - R(1, 0)) / (4 * q(3));
190  }
191 
192  if (q(3) < 0)
193  q = -q;
194  quaternionNormalize(q);
195  return q;
196 }
