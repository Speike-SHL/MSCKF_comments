294  template <typename T>
295  void removeUnmarkedElements(
296  const std::vector<T> &raw_vec,
297  const std::vector<unsigned char> &markers,
298  std::vector<T> &refined_vec)
299  {
300  if (raw_vec.size() != markers.size())
301  {
302  ROS_WARN("The input size of raw_vec(%lu) and markers(%lu) does not match...",
303  raw_vec.size(), markers.size());
304  }
305  for (int i = 0; i < markers.size(); ++i)
306  {
307  if (markers[i] == 0)
308  continue;
309  refined_vec.push_back(raw_vec[i]);
310  }
311  return;
312  }
