162  bool Feature::checkMotion(
163  const CamStateServer &cam_states) const
164  {
165  // 1. 取出单个特征点对应的始末帧id，即单个特征点被观测到的第一帧和最后一帧
166  const StateIDType &first_cam_id = observations.begin()->first;
167  const StateIDType &last_cam_id = (--observations.end())->first;
168 
169  // 2. 分别赋值位姿
170  // 左相机第一帧的Twc
171  Eigen::Isometry3d first_cam_pose;
172  // 左相机第一帧的Rcw -> Rwc
173  first_cam_pose.linear() =
174  quaternionToRotation(cam_states.find(first_cam_id)->second.orientation).transpose();
175  // 左相机第一帧的twc
176  first_cam_pose.translation() =
177  cam_states.find(first_cam_id)->second.position;
178  // 左相机最后一帧的Twc
179  Eigen::Isometry3d last_cam_pose;
180  // 左相机最后一帧的Rcw -> Rwc
181  last_cam_pose.linear() =
182  quaternionToRotation(cam_states.find(last_cam_id)->second.orientation).transpose();
183  // 左相机最后一帧的twc
184  last_cam_pose.translation() =
185  cam_states.find(last_cam_id)->second.position;
186 
187  // Get the direction of the feature when it is first observed.
188  // This direction is represented in the world frame.
189  // 3. 求出投影射线在世界坐标系下的方向
190  Eigen::Vector3d feature_direction(
191  observations.begin()->second(0),
192  observations.begin()->second(1), 1.0); // 特征在第一帧的左相机上的归一化坐标
193  feature_direction = feature_direction / feature_direction.norm(); // 特征在第一帧的左相机上的方向，从相机原点指向归一化坐标的射线，相机系中
194  feature_direction = first_cam_pose.linear() * feature_direction; // 特征在第一帧的世界坐标系下的方向
195 
196  // Compute the translation between the first frame
197  // and the last frame. We assume the first frame and
198  // the last frame will provide the largest motion to
199  // speed up the checking process.
200  // 4. 求出始末两帧在世界坐标系下的位移（这段判断非常精彩！！！！）
201  Eigen::Vector3d translation =
202  last_cam_pose.translation() - first_cam_pose.translation();
203 
204  // 这里相当于两个向量点乘 这个结果等于两个向量的模乘以cos夹角
205  // 也相当于translation 在 feature_direction上的投影
206  // 其实就是translation在feature_direction方向上的长度
207  double parallel_translation =
208  translation.transpose() * feature_direction;
209 
210  // 这块直接理解比较抽象，假设十四讲图7-11中，左图为第一帧，右图为最后一帧，
211  // O1p1为投影射线在世界系下的方向，因为为单位向量，长度为1
212  // O1O2为始末两帧的位移即translation，假设长度为t
213  // 那么parallel_translation就是O1O2在O1p1上的投影，即O1O2在O1p1上的长度，假设O1p1与O1O2的夹角为0，等于t * cosθ
214  // 使用带入法，分别带入 0° 180° 跟90°
215  // 0°，位移方向与射线方向相同，translation方向与feature_direction方向相同， parallel_translation = t, orthogonal_translation = 0
216  // 180°，位移方向与射线方向相反，translation方向与feature_direction方向相反， parallel_translation = -t, orthogonal_translation = 0
217  // 90°，位移方向与射线方向垂直，translation方向与feature_direction方向垂直， parallel_translation = 0, orthogonal_translation = translation
218  // 所以这块的判断即考虑了角度，同时考虑了位移。运动方向要尽可能与投影射线垂直，且位移要足够大
219  Eigen::Vector3d orthogonal_translation =
220  translation - parallel_translation * feature_direction;
221 
222  if (orthogonal_translation.norm() >
223  optimization_config.translation_threshold)
224  return true;
225  else
226  return false;
227  }

229  bool Feature::initializePosition(
230  const CamStateServer &cam_states)
231  {
232  // 存放每个观测以及每个对应相机的pos，注意这块是左右目独立存放，即一帧图像会有两个cam_poses和两个measurements
233  std::vector<Eigen::Isometry3d, Eigen::aligned_allocator<Eigen::Isometry3d>> cam_poses(0);
234  std::vector<Eigen::Vector2d, Eigen::aligned_allocator<Eigen::Vector2d>> measurements(0);
235 
236  // 1. 准备数据，左右相机的归一化坐标和位姿
237  for (auto &m : observations)
238  {
239  // 1.0 在cam_states中查找观测对应的相机位姿
240  auto cam_state_iter = cam_states.find(m.first);
241  if (cam_state_iter == cam_states.end())
242  continue;
243 
244  // 1.1 添加左相机和右相机的归一化坐标
245  measurements.push_back(m.second.head<2>());
246  measurements.push_back(m.second.tail<2>());
247 
248  // 左右相机的 Twc
249  Eigen::Isometry3d cam0_pose;
250  cam0_pose.linear() =
251  quaternionToRotation(cam_state_iter->second.orientation).transpose(); // Rcw -> Rwc
252  cam0_pose.translation() = cam_state_iter->second.position; // twc
253  Eigen::Isometry3d cam1_pose;
254  cam1_pose = cam0_pose * CAMState::T_cam0_cam1.inverse();
255 
256  // 1.2 添加相机位姿
257  cam_poses.push_back(cam0_pose);
258  cam_poses.push_back(cam1_pose);
259  }
260 
261  // All camera poses should be modified such that it takes a
262  // vector from the first camera frame in the buffer to this
263  // camera frame.
264  // 2. 转换坐标系，将所有的相机位姿转换为，第一个相机到当前相机的位姿
265  Eigen::Isometry3d T_c0_w = cam_poses[0]; // 第一帧位姿
266  for (auto &pose : cam_poses)
267  pose = pose.inverse() * T_c0_w; // Twci' * Twc0 = Tciw * Twc0 = Tcic0，代表第一个相机到当前相机的位姿
268 
269  // 3. 使用首末位姿粗略计算出一个三维点坐标
270  Eigen::Vector3d initial_position(0.0, 0.0, 0.0);
271  generateInitialGuess( // 传入的是：TcLastc0(第一帧在最后一帧下的位姿), z0(第一帧的观测), zlast(最后一帧的观测), p(第一帧的三维点)
272  cam_poses[cam_poses.size() - 1], measurements[0],
273  measurements[measurements.size() - 1], initial_position);
274  // 弄成逆深度形式，为了避免局部极小值，提高数值稳定性
275  Eigen::Vector3d solution(
276  initial_position(0) / initial_position(2),
277  initial_position(1) / initial_position(2),
278  1.0 / initial_position(2));
279 
280  // Apply Levenberg-Marquart method to solve for the 3d position.
281  double lambda = optimization_config.initial_damping;
282  int inner_loop_cntr = 0;
283  int outer_loop_cntr = 0;
284  bool is_cost_reduced = false;
285  double delta_norm = 0;
286 
287  // 4. 利用初计算的点计算在各个相机下的误差，作为初始误差
288  double total_cost = 0.0;
289  for (int i = 0; i < cam_poses.size(); ++i)
290  {
291  double this_cost = 0.0;
292  // 计算投影误差（归一化坐标）
293  cost(cam_poses[i], solution, measurements[i], this_cost);
294  total_cost += this_cost;
295  }
296 
297  // Outer loop.
298  // 5. LM优化开始， 优化三维点坐标，不优化位姿，比较简单
299  do
300  {
301  // A是 J^t * J B是 J^t * r
302  // 可能有同学疑问自己当初学的时候是 -J^t * r
303  // 这个无所谓，因为这里是负的更新就是正的，而这里是正的，所以更新是负的
304  // 总之就是有一个是负的，总不能误差越来越大吧
305  Eigen::Matrix3d A = Eigen::Matrix3d::Zero();
306  Eigen::Vector3d b = Eigen::Vector3d::Zero();
307 
308  // 5.1 计算AB矩阵
309  for (int i = 0; i < cam_poses.size(); ++i)
310  {
311  Eigen::Matrix<double, 2, 3> J;
312  Eigen::Vector2d r;
313  double w;
314 
315  // 计算一目相机观测的雅可比与误差
316  // J 归一化坐标误差相对于三维点的雅可比
317  // r
318  // w 权重，同信息矩阵
319  jacobian(cam_poses[i], solution, measurements[i], J, r, w);
320 
321  // 鲁棒核约束
322  if (w == 1)
323  {
324  A += J.transpose() * J;
325  b += J.transpose() * r;
326  }
327  else
328  {
329  double w_square = w * w;
330  A += w_square * J.transpose() * J;
331  b += w_square * J.transpose() * r;
332  }
333  }
334 
335  // Inner loop.
336  // Solve for the delta that can reduce the total cost.
337  // 这层是在同一个雅可比下多次迭代，如果效果不好说明需要调整阻尼因子了，因为线性化不是很好
338  // 如果多次一直误差不下降，退出循环重新计算雅可比
339  do
340  {
341  // LM算法中的lambda
342  Eigen::Matrix3d damper = lambda * Eigen::Matrix3d::Identity();
343  Eigen::Vector3d delta = (A + damper).ldlt().solve(b);
344  // 更新
345  Eigen::Vector3d new_solution = solution - delta;
346  // 统计本次修改量的大小，如果太小表示接近目标值或者陷入局部极小值，那么就没必要继续了
347  delta_norm = delta.norm();
348 
349  // 计算更新后的误差
350  double new_cost = 0.0;
351  for (int i = 0; i < cam_poses.size(); ++i)
352  {
353  double this_cost = 0.0;
354  cost(cam_poses[i], new_solution, measurements[i], this_cost);
355  new_cost += this_cost;
356  }
357 
358  // 如果更新后误差比之前小，说明确实是往好方向发展
359  // 我们高斯牛顿的JtJ比较接近真实情况所以减少阻尼，增大步长，delta变大，加快收敛
360  if (new_cost < total_cost)
361  {
362  is_cost_reduced = true;
363  solution = new_solution;
364  total_cost = new_cost;
365  lambda = lambda / 10 > 1e-10 ? lambda / 10 : 1e-10;
366  }
367  // 如果不行，那么不要这次迭代的结果
368  // 说明高斯牛顿的JtJ不接近二阶的海森矩阵
369  // 那么增大阻尼，减小步长，delta变小
370  // 并且算法接近一阶的最速下降法
371  else
372  {
373  is_cost_reduced = false;
374  lambda = lambda * 10 < 1e12 ? lambda * 10 : 1e12;
375  }
376 
377  } while (inner_loop_cntr++ <
378  optimization_config.inner_loop_max_iteration &&
379  !is_cost_reduced);
380 
381  inner_loop_cntr = 0;
382 
383  // 直到迭代次数到了或者更新量足够小了
384  } while (outer_loop_cntr++ <
385  optimization_config.outer_loop_max_iteration &&
386  delta_norm > optimization_config.estimation_precision);
387 
388  // Covert the feature position from inverse depth
389  // representation to its 3d coordinate.
390  // 取出最后的结果
391  Eigen::Vector3d final_position(solution(0) / solution(2),
392  solution(1) / solution(2), 1.0 / solution(2));
393 
394  // Check if the solution is valid. Make sure the feature
395  // is in front of every camera frame observing it.
396  // 6. 深度验证，把三维点在第一帧相机下的坐标转换回第i帧相机下的坐标，然后判断z是否大于0
397  bool is_valid_solution = true;
398  for (const auto &pose : cam_poses)
399  {
400  Eigen::Vector3d position =
401  pose.linear() * final_position + pose.translation();
402  if (position(2) <= 0)
403  {
404  is_valid_solution = false;
405  std::cout << "失败" << std::endl;
406  break;
407  }
408  }
409 
410  // 7. 更新结果
411  // Convert the feature position to the world frame.
412  position = T_c0_w.linear() * final_position + T_c0_w.translation();
413 
414  if (is_valid_solution)
415  is_initialized = true;
416 
417  return is_valid_solution;
418  }

427  void Feature::cost(
428  const Eigen::Isometry3d &T_c0_ci,
429  const Eigen::Vector3d &x, const Eigen::Vector2d &z,
430  double &e) const
431  {
432  // Compute hi1, hi2, and hi3 as Equation (37).
433  const double &alpha = x(0);
434  const double &beta = x(1);
435  const double &rho = x(2);
436 
437  // [x/z]
438  // h = Rcic0 * [y/z] + 1/z * tcic0 = (Rcic0 * P + tcic0) / z
439  // [ 1 ]
440  // =>
441  // h1 | R11 R12 R13 alpha / rho t1 |
442  // h2 = | R21 R22 R23 * beta / rho + t2 | * rho
443  // h3 | R31 R32 R33 1 / rho t3 |
444  Eigen::Vector3d h =
445  T_c0_ci.linear() * Eigen::Vector3d(alpha, beta, 1.0) +
446  rho * T_c0_ci.translation();
447  double &h1 = h(0);
448  double &h2 = h(1);
449  double &h3 = h(2);
450 
451  // 求出在另一个相机ci下的归一化坐标
452  Eigen::Vector2d z_hat(h1 / h3, h2 / h3);
453 
454  // Compute the residual.
455  // 两个归一化坐标算误差
456  e = (z_hat - z).squaredNorm();
457  return;
458  }

460  void Feature::jacobian(
461  const Eigen::Isometry3d &T_c0_ci,
462  const Eigen::Vector3d &x, const Eigen::Vector2d &z,
463  Eigen::Matrix<double, 2, 3> &J, Eigen::Vector2d &r,
464  double &w) const
465  {
466 
467  // Compute hi1, hi2, and hi3 as Equation (37).
468  const double &alpha = x(0); // x/z
469  const double &beta = x(1); // y/z
470  const double &rho = x(2); // 1/z
471 
472  // [x/z]
473  // h = Rcic0 * [y/z] + 1/z * tcic0 = (Rcic0 * P + tcic0) / z
474  // [ 1 ]
475  // =>
476  // h1 | R11 R12 R13 alpha / rho t1 |
477  // h2 = | R21 R22 R23 * beta / rho + t2 | * rho
478  // h3 | R31 R32 R33 1 / rho t3 |
479  Eigen::Vector3d h = T_c0_ci.linear() * Eigen::Vector3d(alpha, beta, 1.0) +
480  rho * T_c0_ci.translation();
481  double &h1 = h(0);
482  double &h2 = h(1);
483  double &h3 = h(2);
484 
485  // Compute the Jacobian.
486  // 首先明确一下误差与三维点的关系
487  // 下面的r是误差 r = z_hat - z; Eigen::Vector2d z_hat(h1 / h3, h2 / h3)
488  // 我们要求r对三维点的雅可比，其中z是观测，与三维点坐标无关
489  // 因此相当于求z_hat相对于 alpha beta rho的雅可比，首先要求出他们之间的关系
490  // 归一化坐标设为x y
491  // x = h1/h3 y = h2/h3
492  // 因此雅可比矩阵为:
493  // | ∂x/∂alpha ∂x/∂beta ∂x/∂rho |
494  // | ∂y/∂alpha ∂y/∂beta ∂y/∂rho |
495  // 先写出h与alpha beta rho的关系，也就是上面写的
496  // h1 = R11 * alpha + R12 * beta + R13 + rho * t1
497  // h2 = R21 * alpha + R22 * beta + R23 + rho * t2
498  // h3 = R31 * alpha + R32 * beta + R33 + rho * t3
499  // 然后求h1 相对于alpha beta rho的导数 再求h2的 再求h3的
500  // 链式求导法则
501  // ∂x/∂alpha = ∂x/∂h1 * ∂h1/∂alpha + ∂x/∂h3 * ∂h3/∂alpha
502  // ∂x/∂h1 = 1/h3 ∂h1/∂alpha = R11
503  // ∂x/∂h3 = -h1/h3^2 ∂h3/∂alpha = R31
504  // 剩下的就好求了
505 
506  // | R11 R12 t1 |
507  // W = | R21 R22 t2 |
508  // | R31 R32 t3 |
509  Eigen::Matrix3d W;
510  W.leftCols<2>() = T_c0_ci.linear().leftCols<2>();
511  W.rightCols<1>() = T_c0_ci.translation();
512 
513  // h1 / h3 相对于 alpha beta rho的
514  J.row(0) = 1 / h3 * W.row(0) - h1 / (h3 * h3) * W.row(2);
515  // h1 / h3 相对于 alpha beta rho的
516  J.row(1) = 1 / h3 * W.row(1) - h2 / (h3 * h3) * W.row(2);
517 
518  // 求取误差
519  Eigen::Vector2d z_hat(h1 / h3, h2 / h3);
520  r = z_hat - z;
521 
522  // Compute the weight based on the residual.
523  // 使用鲁棒核函数约束
524  double e = r.norm();
525  if (e <= optimization_config.huber_epsilon)
526  w = 1.0;
527  // 如果误差大于huber_epsilon但是没超过他的2倍，2 * huber_epsilon / e ∈ 2 * (1~1/2) 那么会放大权重w>1
528  // 如果误差大的离谱，超过他的2倍，缩小他的权重, 2 * huber_epsilon / e ∈ 2 * (1/2 ~ 0) 那么会缩小权重w<1
529  else
530  w = std::sqrt(2.0 * optimization_config.huber_epsilon / e);
531 
532  return;
533  }

542  void Feature::generateInitialGuess(
543  const Eigen::Isometry3d &T_c1_c2, const Eigen::Vector2d &z1,
544  const Eigen::Vector2d &z2, Eigen::Vector3d &p) const
545  {
546  // 列出方程
547  // P2 = R21 * P1 + t21 下面省略21， 求P1
548  // 两边左乘P2的反对称矩阵
549  // P2^ * (R * P1 + t) = 0
550  // 其中左右可以除以P2的深度，这样P2就成了z2，且P1可以分成z1（归一化平面） 乘上我们要求的深度d
551  // z2^ * (R * z1 * d + t) = 0
552  // 令m = R * z1
553  // z2^ * (m * d + t) = 0
554  // | 0 -1 z2y | ( | m0 | )
555  // | 1 0 -z2x | * ( | m1 | * d + t ) = 0
556  // |-z2y z2x 0 | ( | m2 | )
557  // z2^ * m * d = - z2^ * t
558  // A * d = b
559  // 会发现这三行里面两行是线性相关的，所以只取前两行
560  // 构造一个最小二乘问题去求解深度
561  Eigen::Vector3d m = T_c1_c2.linear() * Eigen::Vector3d(z1(0), z1(1), 1.0);
562 
563  Eigen::Vector2d A(0.0, 0.0);
564  A(0) = m(0) - z2(0) * m(2); // 对应第二行
565  // 按照上面推导这里应该是负的但是不影响，因为我们下边b(1)也给取负了
